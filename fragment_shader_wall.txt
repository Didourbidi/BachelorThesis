#version 420 core

in vec3 param_world_normal; 
in vec3 param_world_pos; 

out vec4 fColor;

uniform struct Light
{
	vec3 pos;	//Light Position in eye-coords
} light;

void main(void)
{
	vec3 light_dir = normalize( vec3(-1) );
	vec3 ambient_color = vec3(0.4f);	
	vec3 Normal = normalize(param_world_normal);

	fColor = vec4 (ambient_color + max( 0, 0.6f * dot( -light_dir, param_world_normal ) ), 1.0f);
	//fColor = vec4 (1, 0, 0, 1);
}
